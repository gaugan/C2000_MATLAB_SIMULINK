12void init_eCAN_B ( uint16_T bitRatePrescaler, uint16_T timeSeg1, uint16_T



13                  timeSeg2, uint16_T sbg, uint16_T sjw, uint16_T sam)



14{



15  uint32_t ui32RegValue;



16  uint16_t ui16CANCTL;



17  EALLOW;



18  CpuSysRegs.PCLKCR10.bit.CAN_B = 1;



19  GpioCtrlRegs.GPAPUD.bit.GPIO12 = 0;



20  GpioCtrlRegs.GPAGMUX1.bit.GPIO12 = 0;



21  GpioCtrlRegs.GPAMUX1.bit.GPIO12 = 2;



22  GpioCtrlRegs.GPAQSEL2.bit.GPIO17 = 3;



23  GpioCtrlRegs.GPAPUD.bit.GPIO17 = 0;



24  GpioCtrlRegs.GPAGMUX2.bit.GPIO17 = 0;



25  GpioCtrlRegs.GPAMUX2.bit.GPIO17 = 2;



26  EDIS;



27  ui32RegValue = (((uint32_T)((bitRatePrescaler-1) & 0x03C0))<<10)|((timeSeg2-1)<<



28    12)|((timeSeg1-1)<<8)|((sjw-1)<<6)|((bitRatePrescaler-1) & 0x3F);



29  CANInit(CANB_BASE);



30  CANClkSourceSelect(CANB_BASE, 0);



31



32  //Set init mode



33  ui16CANCTL = HWREGH(CANB_BASE + CAN_O_CTL);



34  HWREGH(CANB_BASE + CAN_O_CTL) = ui16CANCTL | CAN_CTL_INIT | CAN_CTL_CCE;



35



36  //Set Bit Timing register



37  HWREGH(CANB_BASE + CAN_O_BTR) = (ui32RegValue &



38    CAN_REG_WORD_MASK);



39  HWREGH(CANB_BASE + CAN_O_BTR + 2) = (ui32RegValue >> 16);



40



41  // Restore the saved CAN Control register.



42  HWREGH(CANB_BASE + CAN_O_CTL) = ui16CANCTL;



43  HWREGH(CANB_BASE + CAN_O_IP_MUX21) = 0;



44  HWREGH(CANB_BASE + CAN_O_IP_MUX21 + 2) = 0;



45  CANIntEnable(CANB_BASE, CAN_INT_IE0);



46  CANGlobalIntEnable(CANB_BASE, CAN_GLB_INT_CANINT0);



47



48  // Enable the CAN for operation.



49  CANEnable(CANB_BASE);



50}
