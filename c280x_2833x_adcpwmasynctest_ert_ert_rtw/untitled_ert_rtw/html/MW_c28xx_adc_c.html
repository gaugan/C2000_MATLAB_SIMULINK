8void config_ADCA_SOC0()



9{



10  EALLOW;



11  AdcaRegs.ADCSOC0CTL.bit.CHSEL = 0;   /* Set SOC0 channel select to ADCIN0*/



12  AdcaRegs.ADCSOC0CTL.bit.TRIGSEL = 1;



13  AdcaRegs.ADCSOC0CTL.bit.ACQPS = 6;   /* Set SOC0 S/H Window to 7 ADC Clock Cycles*/



14  AdcaRegs.ADCINTSOCSEL1.bit.SOC0 = 0; /* SOCx No ADCINT Interrupt Trigger Select.*/



15  AdcaRegs.ADCOFFTRIM.bit.OFFTRIM = AdcaRegs.ADCOFFTRIM.bit.OFFTRIM;/* Set Offset Error Correctino Value*/



16  AdcaRegs.ADCCTL1.bit.INTPULSEPOS = 1;/* Late interrupt pulse trips AdcResults latch*/



17  AdcaRegs.ADCSOCPRICTL.bit.SOCPRIORITY = 0;/* All in round robin mode SOC Priority*/



18  EDIS;



19}

21void InitAdcA()



22{



23  EALLOW;



24  CpuSysRegs.PCLKCR13.bit.ADC_A = 1;



25  AdcaRegs.ADCCTL2.bit.PRESCALE = 0;



26  AdcSetMode(ADC_ADCA, ADC_RESOLUTION_12BIT, ADC_SIGNALMODE_SINGLE);



27



28  //power up the ADC



29  AdcaRegs.ADCCTL1.bit.ADCPWDNZ = 1;



30



31  //delay for 1ms to allow ADC time to power up



32  DELAY_US(1000);



33  EDIS;



34}
